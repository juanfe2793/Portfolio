```tsx
import type {ReactNode} from 'react';
import clsx from 'clsx';
import styles from './styles.module.css';

type SkillBadgeProps = {
  name: string;
  level: 'beginner' | 'intermediate' | 'expert';
  category?: string;
};

const CATEGORY_COLORS: Record<string, {light: string; dark: string}> = {
  networking: {light: '#14b8a6', dark: '#2dd4bf'},
  cloud: {light: '#f97316', dark: '#fb923c'},
  orchestration: {light: '#3b82f6', dark: '#60a5fa'},
  observability: {light: '#22c55e', dark: '#4ade80'},
  iac: {light: '#a855f7', dark: '#c084fc'},
  default: {light: '#6b7280', dark: '#9ca3af'},
};

const LEVEL_DOTS = {
  expert: 3,
  intermediate: 2,
  beginner: 1,
} as const;

export default function SkillBadge({name, level, category = 'default'}: SkillBadgeProps): ReactNode {
  const colors = CATEGORY_COLORS[category] ?? CATEGORY_COLORS.default;
  const filledDots = LEVEL_DOTS[level];
  const totalDots = 3;

  return (
    <span
      className={styles.skillBadge}
      style={{
        ['--badge-color' as string]: colors.light,
      }}>
      <span className={styles.badgeName}>{name}</span>
      <span className={styles.dots} aria-label={`${level} level`}>
        {Array.from({length: totalDots}, (_, i) => (
          <span
            key={i}
            className={clsx(styles.dot, i < filledDots ? styles.filled : styles.empty)}
            style={{['--dot-color' as string]: colors.light}}
          />
        ))}
      </span>
    </span>
  );
}
```

```tsx
import type {ReactNode} from 'react';
import styles from './styles.module.css';

export type TimelineEntry = {
  period: string;
  role: string;
  company: string;
  achievements?: string[];
};

type TimelineProps = {
  entries: TimelineEntry[];
};

export default function Timeline({entries}: TimelineProps): ReactNode {
  return (
    <div className={styles.timeline}>
      {entries.map((entry, idx) => (
        <div key={idx} className={styles.timelineEntry}>
          <div className={styles.timelineLeft}>
            <span className={styles.periodBadge}>{entry.period}</span>
            <div className={styles.connector}>
              <div className={styles.line} />
              <div className={styles.node} />
            </div>
          </div>
          <div className={styles.timelineContent}>
            <h3 className={styles.roleTitle}>{entry.role}</h3>
            <p className={styles.companyName}>{entry.company}</p>
            {entry.achievements && entry.achievements.length > 0 && (
              <ul className={styles.achievements}>
                {entry.achievements.map((ach, i) => (
                  <li key={i}>{ach}</li>
                ))}
              </ul>
            )}
          </div>
        </div>
      ))}
    </div>
  );
}
```
